
;

;
; list

define (list . l) l

;
; foldr

define
  foldr fu la li
  if (empty? li)
    la
    fu (car li) (foldr fu la (cdr li))

;
; foldl

define
  foldl fu ac li
  if (empty? li)
    ac
    foldl fu (fu (car li) ac) (cdr li)

;

define
  map func li
  foldr (lambda (x y) (cons (func x) y)) '() li
