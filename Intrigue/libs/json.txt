define (make-jarray . l) l

;
; make-jobject
;

define
  make-jobject . l
  let
    _
      dictionary l
    define
      get dic key
      if (empty? dic)
        '()
        if (equal? (car dic) key)
          car (cdr dic)
          get (cdr (cdr dic)) key
    define
      dispatch meth . args
      ;(apply (environment-lookup meth) (cons dictionary args))
      (apply meth (cons (quote dictionary) args))
    dispatch
