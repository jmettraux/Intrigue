define
  gcd2 m n
  abs
    if (zero? n)
      m                       ; then
      gcd2 n (remainder m n)  ; else

define
  gcd . ns
  if (empty? ns)
    0                                   ; then
    gcd2 (car ns) (apply gcd (cdr ns))  ; else
